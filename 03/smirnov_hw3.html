class MyDecisionTreeClassifier:
    NON_LEAF_TYPE = 0
    LEAF_TYPE = 1

    def __init__(self, min_samples_split=2, max_depth=None, criterion='gini'):
        """
        criterion -- критерий расщепления. необходимо релизовать три:
        ошибка классификации - 'classification_error',
        индекс Джини - 'gini',
        энтропийный критерий - 'entropy',
        max_depth -- максимальная глубина дерева,
        min_samples_split -- минимальное число объектов в листе, чтобы сделать новый сплит
        """
        self.min_samples_split = min_samples_split
        self.max_depth = max_depth
        self.classes = None
        self.num_class = -1
        # Для последнего задания
        self.impurities = defaultdict(int)
        self.feature_importances_ = defaultdict(int)
        self.criterion = criterion
        # Структура, которая описывает дерево
        # Представляет словарь, где для  node_id (айдишник узла дерева) храним
        # (тип_узла, айдишник признака сплита, порог сплита) если тип NON_LEAF_TYPE
        # (тип_узла, предсказание класса, вероятность класса) если тип LEAF_TYPE
        # Подразумевается, что у каждого node_id в дереве слева 
        # узел с айди 2 * node_id + 1, а справа 2 * node_id + 2
        self.tree = dict()

    def __div_samples(self, x, y, feature_id, threshold):
        """
        Разделяет объекты на 2 множества
        x -- матрица объектов
        y -- вектор ответов
        feature_id -- айдишник признака, по которому делаем сплит
        threshold -- порог, по которому делаем сплит
        """
        left_mask = x[:, feature_id] > threshold
        right_mask = ~left_mask
        return x[left_mask], x[right_mask], y[left_mask], y[right_mask]        
    
    
    def __find_threshold(self, x, y):
        """
        Находим оптимальный признак и порог для сплита
        Здесь используемые разные impurity в зависимости от self.criterion
        x -- вектор-признак
        y -- целевая переменная
        """
        indexes_sort = x.argsort()
        sorted_x, sorted_y = x[indexes_sort], y[indexes_sort]
        # Находим индексы, в которых значение признака меняется
        feature_borders = np.where(sorted_x[:-1] != sorted_x[1:])[0] + 1
        # Если значения признака не меняются нигде, то он бесполезен
        if not feature_borders.size:
            return None, -np.inf, None, None
        # Cтроим матрицу one-hot кодирования целевой переменной 
        # (как для обычной категориальной переменной) для каждого объекта выборки
        one_hot = np.zeros((y.size, self.num_class))
        one_hot[np.arange(sorted_y.size), sorted_y] = 1
        # Получаем матрицу долей классов при увеличении значения признака
        all_counts = np.cumsum(one_hot, axis = 0)
        # Доли классов для левого и правого листов в индексах, в которых значение признака меняется
        l_counts = all_counts[feature_borders - 1]
        r_counts = all_counts[-1] - l_counts
        # Количество элементов всего в левом и правом листе в тех же индексах
        l_num = feature_borders
        r_num = y.size - feature_borders
        # Считаем критерий информативности для каждого уникального порога
        # Пояснение корректности такого упрощения представлено ниже
        if self.criterion == 'gini':
            left = np.sum(l_counts ** 2, axis = 1) / l_num
            right = np.sum(r_counts ** 2, axis = 1) / r_num
        elif self.criterion == 'entropy':
            l_probas = l_counts / l_num.reshape(-1, 1)
            r_probas = r_counts / r_num.reshape(-1, 1)
            left = np.sum(l_probas * np.log(l_probas, where = l_probas != 0), axis = 1) * l_num
            right = np.sum(r_probas * np.log(r_probas, where = r_probas != 0), axis = 1) * r_num
        elif self.criterion == 'classification_error':
            left = np.max(l_counts, axis = 1) 
            right = np.max(r_counts, axis = 1)
        else:
            raise Exception("Invalid criteria")
        scores = left + right
        idx = np.argmax(scores)
        F_l = left[idx]
        F_r = right[idx]
        thr_idx = feature_borders[idx]
        threshold = (sorted_x[thr_idx - 1] + sorted_x[thr_idx]) / 2.0
        return threshold, scores[idx], F_l, F_r
        

    def __fit_node(self, x, y, node_id, depth):
        """
        Делаем новый узел в дереве
        Решаем, терминальный он или нет
        Если нет, то строим левый узел  с айди 2 * node_id + 1
        И правый узел с  айди 2 * node_id + 2
        """
        # Если остался единственный класс, не продолжаем разбиение
        if np.unique(y).size == 1:
            split = False
        # Проверяем, что в узле достаточно объектов для разбиения
        elif y.size >= self.min_samples_split:
            if self.max_depth is not None:
                # Если указана максимальная глубина, проверяем, что она еще не достигнута
                split = depth < self.max_depth
            else:
                split = True
        else:
            split = False
                
        if split:
            # Находим максимальный критерий информативности для каждого признака
            scores = np.asarray([self.__find_threshold(x_i, y) for x_i in x.T])
            idx = scores[:, 1].argmax()
            feature_id = idx
            threshold = scores[idx][0]
            F_l = scores[idx][2]
            F_r = scores[idx][3]
            x_left, x_right, y_left, y_right = self.__div_samples(x, y, feature_id, threshold)
            
            # Вычисление gain
            add = self.criterion != 'entropy'
            if node_id:
                F_curr = self.impurities[(node_id - 1) // 2][1 - node_id % 2]
            else:
                F_curr = 1
            Q = scores[idx][1] / y.size - add + F_curr
            self.feature_importances_[feature_id] += Q
            self.impurities[node_id] = np.array([add - F_l / y_left.size, add - F_r / y_right.size])
            
            self.tree[node_id] = (self.__class__.NON_LEAF_TYPE, feature_id, threshold)
            self.__fit_node(x_left, y_left, 2 * node_id + 1, depth + 1)
            self.__fit_node(x_right, y_right, 2 * node_id + 2, depth + 1)
        else:
            bins = np.bincount(y)
            self.tree[node_id] = (self.__class__.LEAF_TYPE, bins.argmax(), bins.max() / bins.sum())
    
    def fit(self, x, y):
        """
        Рекурсивно строим дерево решений
        Начинаем с корня node_id 0
        """
        #считаем, что классы нумеруются с 0 и без пропусков
        self.num_class = np.unique(y).size
        self.feature_importances_ = np.zeros(x.shape[1])
        self.__fit_node(x, y, 0, 0) 

    def __predict_class(self, x, node_id):
        """
        Рекурсивно обходим дерево по всем узлам,
        пока не дойдем до терминального
        """
        node = self.tree[node_id]
        if node[0] == self.__class__.NON_LEAF_TYPE:
            _, feature_id, threshold = node
            if x[feature_id] > threshold:
                return self.__predict_class(x, 2 * node_id + 1)
            else:
                return self.__predict_class(x, 2 * node_id + 2)
        else:
            return node[1]
        
    def predict(self, X):
        """
        Вызывает predict для всех объектов из матрицы X
        """
        return np.array([self.__predict_class(x, 0) for x in X])
    
    def fit_predict(self, x_train, y_train, predicted_x):
        self.fit(x_train, y_train)
        return self.predict(predicted_x)
    
    def get_feature_importance(self):
        """
        Возвращает важность признаков
        """
        return self.feature_importances_

Упрощение задачи оптимизации¶

На каждом шаге построения дерева необходимо разбить все объекты, находящиеся в данном узле, по какому-то признаку так, чтобы максимизировать критерий информативности:

$$Q(i, t) = F(R_{curr}) - \frac{n_l}{n} \cdot F(R_{left}) - \frac{n_r}{n} \cdot F(R_{right}) \to \max \limits_{i, t},$$

где:\ $i$ - номер признака,\ $t$ - порог для признака с номером $i$,\ $R_{curr}$ - множество объектов, находящихся в текущем узле,\ $R_{left} = \{x \in R_{curr}:\ x[i] <= t\}$,\ $R_{right} = \{x \in R_{curr}:\ x[i] > t\}$,\ $n = |R_{curr}|$,\ $n_l = |R_{left}|$,\ $n_r = |R_{right}|$,\ $F$ - функция для вычисления т.н. impurity: индекс Джини, энтропийный критерий или ошибка классификации.

Заметим, что $F(R_{curr})$, а, следовательно, и $n$ не зависят от $i$ и $t$, значит, задача максимизации может быть заменена задачей минимизации: $$Q'(i, t) = n_l \cdot F(R_{left}) + n_r \cdot F(R_{right}) \to \min \limits_{i, t}.$$

Рассмотрим теперь данную задачу для конкретных функций $F$.

Индекс Джини: $$F(R) = 1 - \sum \limits_{k} p_{R, k}^2,$$\ где $p_{R, k} = \frac{|\{x \in R:\ x_{target} = k\}|}{|R|}$ - доля класса $k$ в данном узле. $$Q'(i, t) = n_l \cdot F(R_{left}) + n_r \cdot F(R_{right}) = $$ $$ = n_l \cdot (1 - \frac{|\{x \in R_{left}:\ x_{target} = k\}|^2}{n_l^2}) + n_r \cdot (1 - \frac{|\{x \in R_{right}:\ x_{target} = k\}|^2}{n_r^2}) = $$ $$ = n_l - \frac{|\{x \in R_{left}:\ x_{target} = k\}|^2}{n_l} + n_r - \frac{|\{x \in R_{right}:\ x_{target} = k\}|^2}{n_r}) = \{n_l + n_r = n\} = $$ $$ = n - \frac{|\{x \in R_{left}:\ x_{target} = k\}|^2}{n_l} - \frac{|\{x \in R_{right}:\ x_{target} = k\}|^2}{n_r} \to \min \limits_{i, t}.$$ Аналогично, $n$ не зависит от $i$ и $t$, поэтому заменяем задачу минимзации на задачу максимизации: $$Q''(i, t) = \frac{|\{x \in R_{left}:\ x_{target} = k\}|^2}{n_l} + \frac{|\{x \in R_{right}:\ x_{target} = k\}|^2}{n_r} \to \max \limits_{i, t}.$$ Связь $Q$ и $Q'$: $$Q = F(R_{curr}) - (\frac{n - Q''}{n}) = \frac{Q''}{n} - 1 + F(R_{curr}).$$

Энтропийный критерий: $$F(R) = - \sum \limits_{k} p_{R, k} \cdot log(p_{R, k}).$$ $$Q'(i, t) = n_l \cdot F(R_{left}) + n_r \cdot F(R_{right}) = $$ $$ = n_l \cdot (-\sum \limits_{k} p_{R_{left}, k} \cdot log(p_{R_{left}, k})) + n_r \cdot (-\sum \limits_{k} p_{R_{right}, k} \cdot log(p_{R_{right}, k})) = $$ $$ = - n_l \cdot \sum \limits_{k} p_{R_{left}, k} \cdot log(p_{R_{left}, k}) - n_r \cdot \sum \limits_{k} p_{R_{right}, k} \cdot log(p_{R_{right}, k}).$$ Заменяем задачу минимзации на задачу максимизации: $$Q''(i, t) = n_l \cdot \sum \limits_{k} p_{R_{left}, k} \cdot log(p_{R_{left}, k}) + n_r \cdot \sum \limits_{k} p_{R_{right}, k} \cdot log(p_{R_{right}, k}) \to \max \limits_{i, t}.$$ Связь $Q$ и $Q'$: $$Q = F(R_{curr}) + \frac{Q''}{n}.$$

Ошибка классификации: $$F(R) = 1 - \max \limits_{k} p_{R, k} = 1 - \max \limits_{k} \frac{|\{x \in R:\ x_{target} = k\}|}{|R|}.$$ $$Q'(i, t) = n_l \cdot F(R_{left}) + n_r \cdot F(R_{right}) = $$ $$ = n_l \cdot (1 - \max \limits_{k} \frac{|\{x \in R_{left}:\ x_{target} = k\}|}{n_l}) + n_r \cdot (1 - \max \limits_{k} \frac{|\{x \in R_{right}:\ x_{target} = k\}|}{n_r}) = $$ $$ = n_l - n_l \cdot \max \limits_{k} \frac{|\{x \in R_{left}:\ x_{target} = k\}|}{n_l}) + n_r - n_r \cdot \max \limits_{k} \frac{|\{x \in R_{right}:\ x_{target} = k\}|}{n_r}) = \{n_l + n_r = n\} = $$ $$ = n - \max \limits_{k} |\{x \in R_{left}:\ x_{target} = k\}| - \max \limits_{k} |\{x \in R_{right}:\ x_{target} = k\}|.$$ Аналогично, $n$ не зависит от $i$ и $t$, поэтому заменяем задачу минимзации на задачу максимизации: $$Q''(i, t) = \max \limits_{k} |\{x \in R_{left}:\ x_{target} = k\}| + \max \limits_{k} |\{x \in R_{right}:\ x_{target} = k\}| \to \max \limits_{i, t}.$$ Связь $Q$ и $Q'$: $$Q = F(R_{curr}) - (\frac{n - Q''}{n}) = \frac{Q''}{n} - 1 + F(R_{curr}).$$

Находим самые важные признаки (2 балла)¶

По построенному дереву легко понять, какие признаки лучше всего помогли решить задачу. Часто это бывает нужно не только для сокращения размерности в данных, но и для лучшего понимания прикладной задачи. Например, Вы хотите понять, какие признаки стоит еще конструировать -- для этого нужно понимать, какие из текущих лучше всего работают в дереве.

Самый простой метод -- посчитать число сплитов, где использовался данные признак. Это не лучший вариант, так как по признаку который принимает всего 2 значения, но который почти точно разделяет выборку, число сплитов будет очень 1, но при этом признак сам очень хороший. В этом задании предлагается для каждого признака считать суммарный gain (в лекции обозначено как Q) при использовании этого признака в сплите. Тогда даже у очень хороших признаков с маленьким число сплитов это значение должно быть довольно высоким.

Реализовать это довольно просто: создаете словарь номер фичи : суммарный гейн и добавляете в нужную фичу каждый раз, когда используете ее при построении дерева.
