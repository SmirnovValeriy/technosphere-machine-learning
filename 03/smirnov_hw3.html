class MyDecisionTreeClassifier:
    NON_LEAF_TYPE = 0
    LEAF_TYPE = 1

    def __init__(self, min_samples_split=2, max_depth=None, criterion='gini'):
        """
        criterion --  .   :
          - 'classification_error',
          - 'gini',
          - 'entropy',
        max_depth --   ,
        min_samples_split --     ,    
        """
        self.min_samples_split = min_samples_split
        self.max_depth = max_depth
        self.classes = None
        self.num_class = -1
        #   
        self.impurities = defaultdict(int)
        self.feature_importances_ = defaultdict(int)
        self.criterion = criterion
        # ,   
        #  ,    node_id (  ) 
        # (_,   ,  )   NON_LEAF_TYPE
        # (_,  ,  )   LEAF_TYPE
        # ,    node_id    
        #    2 * node_id + 1,   2 * node_id + 2
        self.tree = dict()

    def __div_samples(self, x, y, feature_id, threshold):
        """
           2 
        x --  
        y --  
        feature_id --  ,    
        threshold -- ,    
        """
        left_mask = x[:, feature_id] > threshold
        right_mask = ~left_mask
        return x[left_mask], x[right_mask], y[left_mask], y[right_mask]        
    
    
    def __find_threshold(self, x, y):
        """
              
           impurity    self.criterion
        x -- -
        y --  
        """
        indexes_sort = x.argsort()
        sorted_x, sorted_y = x[indexes_sort], y[indexes_sort]
        #  ,     
        feature_borders = np.where(sorted_x[:-1] != sorted_x[1:])[0] + 1
        #      ,   
        if not feature_borders.size:
            return None, -np.inf, None, None
        # C  one-hot    
        # (    )    
        one_hot = np.zeros((y.size, self.num_class))
        one_hot[np.arange(sorted_y.size), sorted_y] = 1
        #        
        all_counts = np.cumsum(one_hot, axis = 0)
        #         ,     
        l_counts = all_counts[feature_borders - 1]
        r_counts = all_counts[-1] - l_counts
        #            
        l_num = feature_borders
        r_num = y.size - feature_borders
        #       
        #      
        if self.criterion == 'gini':
            left = np.sum(l_counts ** 2, axis = 1) / l_num
            right = np.sum(r_counts ** 2, axis = 1) / r_num
        elif self.criterion == 'entropy':
            l_probas = l_counts / l_num.reshape(-1, 1)
            r_probas = r_counts / r_num.reshape(-1, 1)
            left = np.sum(l_probas * np.log(l_probas, where = l_probas != 0), axis = 1) * l_num
            right = np.sum(r_probas * np.log(r_probas, where = r_probas != 0), axis = 1) * r_num
        elif self.criterion == 'classification_error':
            left = np.max(l_counts, axis = 1) 
            right = np.max(r_counts, axis = 1)
        else:
            raise Exception("Invalid criteria")
        scores = left + right
        idx = np.argmax(scores)
        F_l = left[idx]
        F_r = right[idx]
        thr_idx = feature_borders[idx]
        threshold = (sorted_x[thr_idx - 1] + sorted_x[thr_idx]) / 2.0
        return threshold, scores[idx], F_l, F_r
        

    def __fit_node(self, x, y, node_id, depth):
        """
            
        ,    
         ,        2 * node_id + 1
              2 * node_id + 2
        """
        #    ,   
        if np.unique(y).size == 1:
            split = False
        # ,       
        elif y.size >= self.min_samples_split:
            if self.max_depth is not None:
                #    , ,     
                split = depth < self.max_depth
            else:
                split = True
        else:
            split = False
                
        if split:
            #       
            scores = np.asarray([self.__find_threshold(x_i, y) for x_i in x.T])
            idx = scores[:, 1].argmax()
            feature_id = idx
            threshold = scores[idx][0]
            F_l = scores[idx][2]
            F_r = scores[idx][3]
            x_left, x_right, y_left, y_right = self.__div_samples(x, y, feature_id, threshold)
            
            #  gain
            add = self.criterion != 'entropy'
            if node_id:
                F_curr = self.impurities[(node_id - 1) // 2][1 - node_id % 2]
            else:
                F_curr = 1
            Q = scores[idx][1] / y.size - add + F_curr
            self.feature_importances_[feature_id] += Q
            self.impurities[node_id] = np.array([add - F_l / y_left.size, add - F_r / y_right.size])
            
            self.tree[node_id] = (self.__class__.NON_LEAF_TYPE, feature_id, threshold)
            self.__fit_node(x_left, y_left, 2 * node_id + 1, depth + 1)
            self.__fit_node(x_right, y_right, 2 * node_id + 2, depth + 1)
        else:
            bins = np.bincount(y)
            self.tree[node_id] = (self.__class__.LEAF_TYPE, bins.argmax(), bins.max() / bins.sum())
    
    def fit(self, x, y):
        """
           
           node_id 0
        """
        #,     0   
        self.num_class = np.unique(y).size
        self.feature_importances_ = np.zeros(x.shape[1])
        self.__fit_node(x, y, 0, 0) 

    def __predict_class(self, x, node_id):
        """
             ,
            
        """
        node = self.tree[node_id]
        if node[0] == self.__class__.NON_LEAF_TYPE:
            _, feature_id, threshold = node
            if x[feature_id] > threshold:
                return self.__predict_class(x, 2 * node_id + 1)
            else:
                return self.__predict_class(x, 2 * node_id + 2)
        else:
            return node[1]
        
    def predict(self, X):
        """
         predict      X
        """
        return np.array([self.__predict_class(x, 0) for x in X])
    
    def fit_predict(self, x_train, y_train, predicted_x):
        self.fit(x_train, y_train)
        return self.predict(predicted_x)
    
    def get_feature_importance(self):
        """
          
        """
        return self.feature_importances_
